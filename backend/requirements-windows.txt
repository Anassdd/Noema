# Versions are PINNED to the combination verified on the dev machine — the prod
# (locked-down Windows) install must be reproducible, not drift to whatever is
# newest. To upgrade: bump a pin, re-run the test suites, commit the new pin.

# Windows requirements. Identical to requirements.txt EXCEPT the graph store:
# the bundled FalkorDB (graphiti-core[falkordblite]) relies on `redislite`, which is
# Unix-only and cannot be installed on Windows. On Windows we run FalkorDB as an
# external server (Docker) and connect to it with the pure-Python `falkordb` client.
# => set GRAPH_BACKEND=falkor_server in backend/.env (see RUN_ON_WINDOWS.md).

# Provider-swap layer
openai==2.44.0
python-dotenv==1.2.2

# Backend API
fastapi==0.138.2
uvicorn[standard]==0.49.0
python-multipart==0.0.32

# PDF parsing — renders pages to images locally (Windows wheels available).
pypdfium2==5.11.0

# Azure Document Intelligence (only when PARSER=docintel). Pure-Python, cross-platform.
azure-ai-documentintelligence==1.0.2

# RAG vector store — Chroma (Windows wheels available on 1.0+).
chromadb==1.5.9

# Graph memory — Graphiti WITHOUT the falkordblite extra, plus the falkordb client that
# connects to an external FalkorDB server (Docker). This is the Windows-safe combination.
graphiti-core==0.29.2
falkordb==1.6.1
networkx==3.6.1

# LightRAG — the second memory strategy. File-based default storages (networkx graphml +
# nano-vectordb + JSON KV), pure-Python, cross-platform; LLM/embeddings go through our
# provider abstraction, so its own bindings are unused.
lightrag-hku==1.5.4
